  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss.SSS' }),

  winston.format.splat(),
  winston.format.json()

  winston.format.colorize(),
  winston.format.timestamp({ format: 'YYYY-MM-DD HH:mm:ss.SSS' }),
  winston.format.printf(({ timestamp, level, message, ...meta }) => {
    let msg = `${timestamp} [${level}]: ${message}`;
    if (Object.keys(meta).length > 0) {
      msg += ` ${JSON.stringify(meta)}`;
    }
    return msg;
  })
);

  level: process.env.LOG_LEVEL || 'info',
  format: logFormat,
  defaultMeta: { service: appName },

    new winston.transports.Console({
      format: process.env.NODE_ENV === 'production'

      filename: path.join(logDir, `${appName}-%DATE%.log`),
      datePattern: 'YYYY-MM-DD',
      maxSize: '100m',
      maxFiles: '14d',
      zippedArchive: true,
    }),
    
    // é”™è¯¯æ—¥å¿—
    new DailyRotateFile({
      filename: path.join(logDir, `${appName}-error-%DATE%.log`),
      datePattern: 'YYYY-MM-DD',
      level: 'error',
      maxSize: '100m',
      maxFiles: '90d',
      zippedArchive: true,
    }),
    
    // è­¦å‘Šæ—¥å¿—
    new DailyRotateFile({
      filename: path.join(logDir, `${appName}-warn-%DATE%.log`),
      datePattern: 'YYYY-MM-DD',
      level: 'warn',
      maxSize: '100m',
      maxFiles: '30d',
      zippedArchive: true,
    }),
    
    // ä¿¡æ¯æ—¥å¿—
    new DailyRotateFile({
      filename: path.join(logDir, `${appName}-info-%DATE%.log`),
      datePattern: 'YYYY-MM-DD',
      level: 'info',
      maxSize: '100m',
      maxFiles: '14d',
      zippedArchive: true,
    }),
  ],
});

// å¼€å‘ç¯å¢ƒæ·»åŠ è°ƒè¯•æ—¥å¿—

    datePattern: 'YYYY-MM-DD',
    level: 'debug',
    maxSize: '100m',
    maxFiles: '7d',
    zippedArchive: true,
  }));
}
